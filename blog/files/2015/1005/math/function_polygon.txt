//n = number of sides, rotate_pi = polygon direction
void Polygon(int n, float radius, float rotate_pi, float cx, float cy, float cz)
{
  pushMatrix();
    translate(cx,cy,cz);
    float x,y,z;
    beginShape();
      for (int i=0; i<n; i++) {
        x = cos(TWO_PI/n*i)*radius;
        y = sin(TWO_PI/n*i)*radius;
        z=cz;
        vertex(x,y,z);
      }
      rotateZ(rotate_pi);
    endShape(CLOSE);
  popMatrix();
}
